  // ---------- migration ----------

  function migrate(s){

    if(!s) return s;

    var v = s.version || 1;

    if(v < 2){

      var toGround = {"m-event":1,"m-barn":2,"m-light":3,"m-shed":4,"d5":5,"m-supply":6};

      (s.zones||[]).forEach(function(z){ if(toGround[z.id]){ z.level="ground"; z.col=13; z.row=toGround[z.id]; } });

      (s.zones||[]).forEach(function(z){ if(z.level==="mezz"){ if(z.col===3)z.col=1; else if(z.col===4)z.col=2; else if(z.col===5)z.col=3; } });

      v = 2; s.version = 2;

    }

    if(v < 3){

      (s.zones||[]).forEach(function(z){

        if(!z.type){

          if(z.kind==="desk"){ z.type="desk"; z.name=String(z.num); }

          else { z.type=(z.tone==="util")?"other":"conference"; z.name=z.label; }

        }

        if(z.w==null) z.w=1; if(z.h==null) z.h=1;

        if(z.type==="desk" && !z.people) z.people=[];

        delete z.kind; delete z.tone; delete z.num; delete z.label; delete z.open;

      });

      ["ground","mezz"].forEach(function(lv){

        var used = {};

        (s.zones||[]).filter(function(z){return z.level===lv;}).forEach(function(z){ for(var c=z.col;c<z.col+(z.w||1);c++) used[c]=true; });

        var sorted = Object.keys(used).map(Number).sort(function(a,b){return a-b;});

        var map = {}; sorted.forEach(function(c,i){ map[c]=i+1; });

        (s.zones||[]).filter(function(z){return z.level===lv;}).forEach(function(z){ if(map[z.col]) z.col=map[z.col]; });

      });

      v = 3; s.version = 3;

    }

    if(v < 4){ if(s.cell==null) s.cell=96; v=4; s.version=4; }

    return s;

  }

    $("searchClear").addEventListener("click", function(){ $("search").value=""; applySearch(""); $("search").focus(); });

    document.querySelectorAll(".seg [data-view]").forEach(function(b){ b.addEventListener("click", function(){ setView(b.dataset.view); }); });

    $("printBtn").addEventListener("click", function(){ window.print(); });

    $("moveCancel").addEventListener("click", clearSelection);

    $("modePeople").addEventListener("click", function(){ setMode("people"); });

    $("modeLayout").addEventListener("click", function(){ setMode("layout"); });


    $("adminBtn").addEventListener("click", function(){ if(isAdmin){ setAdmin(false); return; } $("adminModal").classList.add("show"); $("adminErr").textContent=""; $("adminCode").value=""; setTimeout(function(){$("adminCode").focus();},50); });

    $("modalCancel").addEventListener("click", function(){ $("adminModal").classList.remove("show"); });

    $("adminModal").addEventListener("click", function(e){ if(e.target===$("adminModal")) $("adminModal").classList.remove("show"); });

    function tryUnlock(){ if($("adminCode").value===ADMIN_CODE){ $("adminModal").classList.remove("show"); setAdmin(true); toast("Editing unlocked"); } else { $("adminErr").textContent="That passcode didn't match. Try again or contact Operations."; } }

    $("modalUnlock").addEventListener("click", tryUnlock);

    $("adminCode").addEventListener("keydown", function(e){ if(e.key==="Enter") tryUnlock(); });


    function add(){ var v=$("newName").value.trim(); if(!v) return; state.unassigned.push({id:"p"+(++pid),name:v}); $("newName").value=""; saveState(); renderAll(); toast(v+" added \u2014 drag to a desk"); $("newName").focus(); }

    $("addBtn").addEventListener("click", add);

    $("newName").addEventListener("keydown", function(e){ if(e.key==="Enter") add(); });


    $("addAreaBtn").addEventListener("click", addArea);

    $("cellMinus").addEventListener("click", function(){ setCell(-8); });

    $("cellPlus").addEventListener("click", function(){ setCell(8); });


    // inspector

    $("inspClose").addEventListener("click", deselectTile);

    $("inspDone").addEventListener("click", deselectTile);

    $("iName").addEventListener("input", function(e){ setName(e.target.value); });

    document.querySelectorAll(".styles button").forEach(function(b){ b.addEventListener("click", function(){ setType(b.dataset.type); }); });

    $("wMinus").addEventListener("click", function(){ setSize(-1,0); });

    $("wPlus").addEventListener("click", function(){ setSize(1,0); });

    $("hMinus").addEventListener("click", function(){ setSize(0,-1); });

    $("hPlus").addEventListener("click", function(){ setSize(0,1); });

    $("mUp").addEventListener("click", function(){ nudge(0,-1); });

    $("mDown").addEventListener("click", function(){ nudge(0,1); });

    $("mLeft").addEventListener("click", function(){ nudge(-1,0); });

    $("mRight").addEventListener("click", function(){ nudge(1,0); });

    $("lvGround").addEventListener("click", function(){ setLevel("ground"); });

    $("lvMezz").addEventListener("click", function(){ setLevel("mezz"); });

    $("delArea").addEventListener("click", deleteArea);


    // export / import / reset

    $("exportBtn").addEventListener("click", function(){ var blob=new Blob([JSON.stringify(state,null,2)],{type:"application/json"}); var a=document.createElement("a"); a.href=URL.createObjectURL(blob); a.download="post-seating-backup-"+new Date().toISOString().slice(0,10)+".json"; a.click(); URL.revokeObjectURL(a.href); });

    $("importBtn").addEventListener("click", function(){ $("importFile").click(); });

    $("importFile").addEventListener("change", function(e){ var f=e.target.files[0]; if(!f) return; var rd=new FileReader(); rd.onload=function(){ try{ var s=migrate(JSON.parse(rd.result)); if(s.zones){ state=s; saveState("Backup restored"); renderAll(); } else toast("That file didn't look right"); }catch(err){ toast("Couldn't read that file"); } }; rd.readAsText(f); e.target.value=""; });

    $("resetBtn").addEventListener("click", function(){ if(confirm("Reset the whole map back to the original layout? This replaces the current arrangement for everyone.")){ deselectTile(); state=defaultState(); saveState("Reset to original layout"); renderAll(); } });


    document.addEventListener("keydown", function(e){ if(e.key==="Escape"){ clearSelection(); deselectTile(); $("adminModal").classList.remove("show"); } });

  }


  // ---------- init ----------

  (async function init(){

    showStoreBanner();

    var loaded=await loadState();

    if(loaded && loaded.zones){ var before=loaded.version||1; state=migrate(loaded); if(before!==4) await saveState(); }

    else { state=defaultState(); await saveState(); }

    var maxp=0; (state.zones||[]).forEach(function(z){ (z.people||[]).forEach(function(p){ var n=parseInt(String(p.id).replace(/\D/g,""),10); if(n>maxp)maxp=n; }); });

    (state.unassigned||[]).forEach(function(p){ var n=parseInt(String(p.id).replace(/\D/g,""),10); if(n>maxp)maxp=n; });

    if(maxp>pid) pid=maxp;

    wire(); setView("both"); renderAll(); updateCellLabel(); startPolling();

  })();

})();
